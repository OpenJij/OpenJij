144  void reset_binaries(const graph::Binaries &init_binaries) {
145 
146  cimod::CheckVariables(init_binaries, vartype);
147 
148  if (init_binaries.size() != binaries.size()) {
149  throw std::runtime_error("The size of initial binaries does not equal to system size");

152  if (binaries[index_binary] != init_binaries[index_binary]) {

155  if (binaries[index_binary] != init_binaries[index_binary]) {

164  void reset_dE() {
165  dE_.clear();
166  dE_v_.clear();
167  dE_.resize(num_binaries);
168  dE_v_.resize(num_binaries);
169 
170  //Initialize
171  max_effective_dE_ = std::abs(poly_value_list_.front());
172 
173  for (const auto &index_binary: active_binaries_) {
174  FloatType val = 0.0;
175  FloatType abs_val = 0.0;
176  bool flag = false;
177  const graph::Binary binary = binaries[index_binary];
178  for (const auto &index_key: adj_[index_binary]) {
179  if (zero_count_[index_key] + binary == 1) {
180  val += poly_value_list_[index_key];
181  }
182  flag = true;
183  abs_val += std::abs(poly_value_list_[index_key]);
184  }
185  dE_[index_binary] = (-2*binary + 1)*val;
186  dE_v_[index_binary] = dE_[index_binary];
187 
188  if (flag && max_effective_dE_ < abs_val) {
189  max_effective_dE_ = abs_val;
190  }
191  }

237  void update_system_single(const graph::Index index_update_binary) {
238  const graph::Binary update_binary = binaries[index_update_binary];
239  const int coeef = -2*update_binary + 1;
240  const int count = +2*update_binary - 1;
241  for (const auto &index_key: adj_[index_update_binary]) {
242  const FloatType val = poly_value_list_[index_key];
243  for (const auto &index_binary: poly_key_list_[index_key]) {
244  const graph::Binary binary = binaries[index_binary];
245  if (zero_count_[index_key] + update_binary + binary == 2 && index_binary != index_update_binary) {
246  dE_[index_binary] += coeef*(-2*binary + 1)*val;
247  dE_v_[index_binary] = dE_[index_binary];
248  }
249  }
250  zero_count_[index_key] += count;
251  zero_count_v_[index_key] = zero_count_[index_key];
252  }
253  dE_[index_update_binary] *= -1;
254  dE_v_[index_update_binary] = dE_[index_update_binary];
255  binaries[index_update_binary] = 1 - binaries[index_update_binary];
256  binaries_v_[index_update_binary] = binaries[index_update_binary];
257  }

261  void virtual_update_system_single(const graph::Index index_update_binary) {
262  const graph::Binary update_binary = binaries_v_[index_update_binary];
263  const int coeef = -2*update_binary + 1;
264  const int count = +2*update_binary - 1;
265  for (const auto &index_key: adj_[index_update_binary]) {
266  const FloatType val = poly_value_list_[index_key];
267  for (const auto &index_binary: poly_key_list_[index_key]) {
268  const graph::Binary binary = binaries_v_[index_binary];
269  if (zero_count_v_[index_key] + update_binary + binary == 2 && index_binary != index_update_binary) {
270  dE_v_[index_binary] += coeef*(-2*binary + 1)*val;
271  update_index_dE_v_.emplace(index_binary);
272  }
273  }
274  zero_count_v_[index_key] += count;
275  update_index_zero_count_v_.push_back(index_key);
276  }
277  dE_v_[index_update_binary] *= -1;
278  update_index_dE_v_.emplace(index_update_binary);
279  binaries_v_[index_update_binary] = 1 - binaries_v_[index_update_binary];
280  update_index_binaries_v_.push_back(index_update_binary);
281  }

475  void SortInteractions() {
476 
477  std::vector<graph::Index> index(num_interactions_);
478 #pragma omp parallel for
479  for (int64_t i = 0; i < num_interactions_; ++i) {
480  index[i] = i;
481  }
482 
483  auto compare_value = [this](const std::size_t i1, const std::size_t i2) {
484  return poly_value_list_[i1] < poly_value_list_[i2];
485  };
486  auto compare_size = [this](const std::size_t i1, const std::size_t i2) {
487  return poly_key_list_[i1].size() < poly_key_list_[i2].size();
488  };
489 
490  std::stable_sort(index.begin(), index.end(), compare_size);
491  std::stable_sort(index.begin(), index.end(), compare_value);
492 
493  cimod::PolynomialValueList<FloatType> vv = poly_value_list_;
494 
495 #pragma omp parallel for
496  for (int64_t i = 0; i < num_interactions_; ++i) {
497  poly_value_list_[i] = vv[index[i]];
498  }
499 
500  cimod::PolynomialValueList<FloatType>().swap(vv);
501 
502  cimod::PolynomialKeyList<graph::Index> ii = poly_key_list_;
503 
504 #pragma omp parallel for
505  for (int64_t i = 0; i < num_interactions_; ++i) {
506  poly_key_list_[i] = ii[index[i]];
507  }
508 
509  }

513  void SetAdj() {
514  adj_.clear();
515  adj_.resize(num_binaries);
516  for (int64_t i = 0; i < num_interactions_; ++i) {
517  for (const auto &index: poly_key_list_[i]) {
518  adj_[index].push_back(i);
519  }
520  }
521 
522  //sort by value and key size
523  auto compare_size = [this](const int64_t i1, const int64_t i2) { return poly_key_list_[i1].size() < poly_key_list_[i2].size(); };
524  auto compare_value = [this](const int64_t i1, const int64_t i2) { return poly_value_list_[i1] < poly_value_list_[i2]; };
525 
526  int64_t adj_size = static_cast<int64_t>(adj_.size());
527 #pragma omp parallel for
528  for (int64_t i = 0; i < adj_size; ++i) {
529  std::stable_sort(adj_[i].begin(), adj_[i].end(), compare_size);
530  std::stable_sort(adj_[i].begin(), adj_[i].end(), compare_value);
531  }
532  }

553  std::pair<std::vector<graph::Index>, FloatType> FindMaxInteraction() const {
554  if (poly_key_list_.size() == 0) {
555  throw std::runtime_error("Interactions are empty.");
556  }
557  FloatType max_val = 0.0;
558  std::vector<graph::Index> max_key = {};
559  for (std::size_t i = 0; i < poly_key_list_.size(); ++i) {
560  if (std::abs(max_val) < std::abs(poly_value_list_[i])) {
561  max_val = poly_value_list_[i];
562  max_key = poly_key_list_[i];
563  }
564  }
565  return std::pair<std::vector<graph::Index>, FloatType>(max_key, max_val);
566  }

570  std::pair<std::vector<graph::Index>, FloatType> FindMinInteraction(const FloatType threshold = 0.0) const {
571  if (poly_key_list_.size() == 0) {
572  throw std::runtime_error("Interactions are empty.");
573  }
574  FloatType min_val = poly_value_list_[0];
575  std::vector<graph::Index> min_key = poly_key_list_[0];
576  for (std::size_t i = 0; i < poly_key_list_.size(); ++i) {
577  if (poly_value_list_[i] != 0.0 && std::abs(poly_value_list_[i]) < std::abs(min_val) && threshold < std::abs(poly_value_list_[i])) {
578  min_val = poly_value_list_[i];
579  min_key = poly_key_list_[i];
580  }
581  }
582 
583  if (std::abs(min_val) <= threshold) {
584  throw std::runtime_error("No minimum value in interactions");
585  }
586  return std::pair<std::vector<graph::Index>, FloatType>(min_key, min_val);
587  }
