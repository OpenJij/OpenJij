174  FloatType energy(const Spins& spins) const{
175  if(spins.size() != this->get_num_spins()){
176  throw std::out_of_range("Out of range in energy in Dense graph.");
177  }
178 
179  using Vec = Eigen::Matrix<FloatType, Eigen::Dynamic, 1, Eigen::ColMajor>;
180  Vec s(get_num_spins()+1);
181  for(size_t i=0; i<spins.size(); i++){
182  s(i) = spins[i];
183  }
184  s(get_num_spins()) = 1;
185 
186  // the energy must be consistent with BinaryQuadraticModel.
187  return (s.transpose()*(_J.template triangularView<Eigen::Upper>()*s))(0,0)-1;
188  }
