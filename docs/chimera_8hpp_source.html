152  inline void _checkpair(Index idx1, Index idx2) const{
153  std::int64_t r1,c1,i1,r2,c2,i2;
154  std::tie(r1,c1,i1) = to_rci(idx1);
155  std::tie(r2,c2,i2) = to_rci(idx2);
156  if(
157  !((r1 == r2 && std::abs(c1 - c2) == 1 && i1 == i2 && i1 >= 4) || //horizontal connect
158  (c1 == c2 && std::abs(r1 - r2) == 1 && i1 == i2 && 4 < i1) || //vertical connect
159  (r1 == r2 && c1 == c2 && i1 < 4 && 4 <= i2) || //in-site connect
160  (r1 == r2 && c1 == c2 && i2 < 4 && 4 <= i1) || //in-site connect
161  (r1 == r2 && c1 == c2 && i1 == i2)) //local field
162  ){
163  throw std::runtime_error("invalid index pair "+std::to_string(idx1)+" "+std::to_string(idx2)+" inserted in Chimera");
164  }
165  }

179  Index to_ind(std::int64_t r, std::int64_t c, std::int64_t i) const{
180  //row-major matrix
181 
182  if(!(-1 <= r && r <= static_cast<std::int64_t>(_num_row))){
183  throw std::runtime_error("invalid value r="+std::to_string(r)+" inserted in Chimera");
184  }
185 
186  if(!(-1 <= c && c <= static_cast<std::int64_t>(_num_column))){
187  throw std::runtime_error("invalid value c="+std::to_string(c)+" inserted in Chimera");
188  }
189 
190  if(!(0 <= i && i < static_cast<std::int64_t>(_num_in_chimera))){
191  throw std::runtime_error("invalid value i="+std::to_string(i)+" inserted in Chimera");
192  }
193 
194  return _num_column*_num_in_chimera * mod_r(r) + _num_in_chimera * mod_c(c) + i;
195  }
196 
204  ChimeraIndex to_rci(Index ind) const{
205  if(!(ind < this->get_num_spins())){
206  throw std::runtime_error("invalid value index="+std::to_string(ind)+" inserted in Chimera");
207  }
208 
209  std::int64_t r = ind/(_num_column*_num_in_chimera);
210  ind -= (_num_column*_num_in_chimera)*r;
211  std::int64_t c = ind/_num_in_chimera;
212  ind -= (_num_in_chimera)*c;
213  std::int64_t i = ind;
214  return std::make_tuple(r, c, i);
215  }

224  Chimera(std::size_t num_row, std::size_t num_column, FloatType init_val=0)
225  : Sparse<FloatType>(num_row*num_column*_num_in_chimera, 6+1), _init_val(init_val), _num_row(num_row), _num_column(num_column){
226  //generate sparse graph
227  assert(_num_row >= 1);
228  assert(_num_column >= 1);
229 
230  for(std::size_t r=0; r<_num_row; r++){
231  for(std::size_t c=0; c<_num_column; c++){
232  for(std::size_t i=0; i<_num_in_chimera; i++){
233  //open boundary
234  if(r > 0 && i < 4){
235  //MINUS_R (0<=i<4)
236  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r-1,c,i)) = _init_val;
237  }
238  if(c > 0 && 4 <= i){
239  //MINUS_C (4<=i<8)
240  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c-1,i)) = _init_val;
241  }
242  if(r < num_row-1 && i < 4){
243  //PLUS_R (0<=i<4)
244  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r+1,c,i)) = _init_val;
245  }
246  if(c < num_column-1 && 4 <= i){
247  //PLUS_C (4<=i<8)
248  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c+1,i)) = _init_val;
249  }
250 
251  //inside chimera unit
252 
253  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c,(i < 4)?4:0)) = _init_val;
254  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c,(i < 4)?5:1)) = _init_val;
255  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c,(i < 4)?6:2)) = _init_val;
256  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c,(i < 4)?7:3)) = _init_val;
257 
258  //local field
259  this->Sparse<FloatType>::J(to_ind(r,c,i), to_ind(r,c,i)) = _init_val;
260  }
261  }
262  }

273  Chimera(const json& j, std::size_t num_row, std::size_t num_column, FloatType init_val=0) : Chimera<FloatType>(num_row, num_column, init_val){
274  if(!(j["num_variables"] <= num_row*num_column*_num_in_chimera)){
275  throw std::runtime_error("number of system size does not match");
276  }
277  //define bqm with ising variables
278  auto bqm = json_parse<FloatType, cimod::Sparse>(j, false);
279  //interactions
280  for(auto&& elem : bqm.get_quadratic()){
281  const auto& key = elem.first;
282  const auto& val = elem.second;
283  _checkpair(key.first, key.second);
284  this->Sparse<FloatType>::J(key.first, key.second) += val;
285  }
286  //local field
287  for(auto&& elem : bqm.get_linear()){
288  const auto& key = elem.first;
289  const auto& val = elem.second;
290  this->Sparse<FloatType>::h(key) += val;
291  }
292  }
