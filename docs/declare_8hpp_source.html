116  .def("get_num_interactions", &Poly::get_num_interactions)

118  .def("energy" , &Poly::energy, "spins"_a, "omp_flag"_a = true)
119  .def("__setitem__" , [](Poly& self, graph::Index key, FloatType val){ self.J(key) = val;}, "key"_a, "val"_a)
120  .def("__setitem__" , [](Poly& self, std::vector<graph::Index>& key, FloatType val){ self.J(key) = val;}, "key"_a, "val"_a)
121  .def("__getitem__" , [](const Poly& self, std::vector<graph::Index>& key){ return self.J(key); }, "key"_a)
122  .def("__getitem__" , [](const Poly& self, graph::Index key){ return self.J(key); }, "key"_a)
123  .def("get_polynomial" , [](const Poly& self) {
124  py::dict py_polynomial;
125  for (std::size_t i = 0; i < self.get_keys().size(); ++i) {
126  py::tuple temp;
127  for (const auto &it: self.get_keys()[i]) {
128  temp = temp + py::make_tuple(it);
129  }
130  py_polynomial[temp] = self.get_values()[i];
131  }
132  return py_polynomial;
133  });
