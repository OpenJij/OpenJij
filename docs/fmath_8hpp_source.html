451  if (x >= 709.78271289338397) return std::numeric_limits<double>::infinity();

460  uint64_t iax = c.tbl[di & mask(c.sbit)];
461  __m128d _t = _mm_sub_sd(_mm_mul_sd(_mm_sub_sd(d, b), _mm_load_sd(&c.ra)), xx);
462  uint64_t u = ((di + c.adj) >> c.sbit) << 52;
463  double t;
464  _mm_store_sd(&t, _t);
465  double y = (c.C3[0] - t) * (t * t) * c.C2[0] - t + c.C1[0];
466  double did;
467  u |= iax;
468  memcpy(&did, &u, sizeof(did));
469  return y * did;
470 #else
471 /*

473 */

475  di di;
476  di.d = x * c.a + b;
477  uint64_t iax = c.tbl[di.i & mask(c.sbit)];
478 
479  double t = (di.d - b) * c.ra - x;
480  uint64_t u = ((di.i + c.adj) >> c.sbit) << 52;
481  double y = (c.C3[0] - t) * (t * t) * c.C2[0] - t + c.C1[0];

801  get f_y(x) := pow(x, y) for x >= 0

803 class PowGenerator {
804  enum {
805  N = 11
806  };
807  float tbl0_[256];
808  struct {
809  float app;
810  float rev;
811  } tbl1_[1 << N];
812 public:
813  PowGenerator(float y)
814  {
815  for (int i = 0; i < 256; i++) {
816  tbl0_[i] = ::powf(2, (i - 127) * y);
817  }
818  const double e = 1 / double(1 << 24);
819  const double h = 1 / double(1 << N);
820  const size_t n = 1U << N;
821  for (size_t i = 0; i < n; i++) {
822  double x = 1 + double(i) / n;
823  double a = ::pow(x, (double)y);
824  tbl1_[i].app = (float)a;
825  double b = ::pow(x + h - e, (double)y);
826  tbl1_[i].rev = (float)((b - a) / (h - e) / (1 << 23));
827  }
828  }
829  float get(float x) const
830  {
831  using namespace local;
832  fi fi;
833  fi.f = x;
834  int a = (fi.i >> 23) & mask(8);
835  unsigned int b = fi.i & mask(23);
836  unsigned int b1 = b & (mask(N) << (23 - N));
837  unsigned int b2 = b & mask(23 - N);
838  float f;
839  int idx = b1 >> (23 - N);
840  f = tbl0_[a] * (tbl1_[idx].app + float(b2) * tbl1_[idx].rev);
841  return f;
842  }
