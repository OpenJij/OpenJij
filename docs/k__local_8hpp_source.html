45  auto urd = std::uniform_real_distribution<>(0, 1.0);
46 
47  int64_t count = 0;
48 
49  for (const auto &index_binary: system.get_active_binaries()) {
50  const FloatType dE_s = system.dE_single(index_binary);
51  if (system.count_call_updater%system.rate_call_k_local == 0 && dE_s == 0.0) {
52  for (const auto &index_key: system.get_adj(index_binary)) {
53  if (system.GetPolyValue(index_key) >= 0.0) {
54  break;
55  }
56  const FloatType dE_i = system.dE_k_local(index_key);
57  if (dE_i <= 0.0|| std::exp(-parameter.beta*dE_i) > urd(random_number_engine)) {
58  system.update_system_k_local();
59  }
60  else {
61  system.reset_virtual_system();
62  }
63  }
64  }
65  else if (dE_s <= 0.0 || std::exp(-parameter.beta*dE_s) > urd(random_number_engine)) {
66  system.update_system_single(index_binary);
67  }
68 
69  if (dE_s >= 0.0) {
70  count++;
71  }
72  }
73 
74  if (count == static_cast<int64_t>(system.get_active_binaries().size()) && system.count_call_updater%system.rate_call_k_local != 0) {
75  for (const auto &index_binary: system.get_active_binaries()) {
76  const FloatType dE_s = system.dE_single(index_binary);
77  if (dE_s == 0.0 && system.binaries[index_binary] == 1) {
78  system.update_system_single(index_binary);
79  }
80  }
81  for (int64_t index_key = 0; index_key < system.GetNumInteractions(); index_key++) {
82  if (system.GetPolyValue(index_key) >= 0.0) {
83  break;
84  }
85  if (system.GetZeroCount(index_key) != 0) {
86  const FloatType dE_i = system.dE_k_local(index_key);
87  if (dE_i < 0.0) {
88  system.update_system_k_local();
89  }
90  else {
91  system.reset_virtual_system();
92  }
93  }
94  }
95  }
96 
97  system.count_call_updater++;
98  }
