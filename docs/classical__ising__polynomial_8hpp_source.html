126  void reset_variables(const graph::Spins &init_variables) {
127  if (init_variables.size() != variables.size()) {
128  throw std::runtime_error("The size of initial spins/binaries does not equal to system size");
129  }
130  cimod::CheckVariables(init_variables, vartype);
131  if (vartype == cimod::Vartype::SPIN) {
132  for (const auto &index_variable: active_variables_) {
133  if (variables[index_variable] != init_variables[index_variable]) {
134  update_spin_system(index_variable);
135  }
136  if (variables[index_variable] != init_variables[index_variable]) {
137  std::stringstream ss;
138  ss << "Unknown error detected in " << __func__;
139  throw std::runtime_error(ss.str());
140  }
141  }
142  }
143  else if (vartype == cimod::Vartype::BINARY) {
144  for (const auto &index_variable: active_variables_) {
145  if (variables[index_variable] != init_variables[index_variable]) {
146  update_binary_system(index_variable);
147  }
148  if (variables[index_variable] != init_variables[index_variable]) {
149  std::stringstream ss;
150  ss << "Unknown error detected in " << __func__;
151  throw std::runtime_error(ss.str());
152  }
153  }
154  }
155  else {
156  throw std::runtime_error("Unknown vartype detected");
157  }
158  }

161  void reset_dE() {
162  dE_.clear();
163  dE_.resize(num_variables);
164 
165  if (vartype == cimod::Vartype::SPIN) {
166  //Initialize
167  max_effective_dE_ = 2.0*std::abs(poly_value_list_.front());
168  for (const auto &index_binary: active_variables_) {
169  FloatType val = 0.0;
170  FloatType abs_val = 0.0;
171  bool flag = false;
172  for (const auto &index_key: adj_[index_binary]) {
173  val += poly_value_list_[index_key]*sign_key_[index_key];
174  abs_val += std::abs(poly_value_list_[index_key]);
175  flag = true;
176  }
177  dE_[index_binary] = -2*val;
178  if (flag && max_effective_dE_ < 2*abs_val) {
179  max_effective_dE_ = 2*abs_val;
180  }
181  }
182  }
183  else if (vartype == cimod::Vartype::BINARY) {
184  //Initialize
185  max_effective_dE_ = std::abs(poly_value_list_.front());
186  for (const auto &index_binary: active_variables_) {
187  FloatType val = 0.0;
188  FloatType abs_val = 0.0;
189  bool flag = false;
190  const graph::Binary binary = variables[index_binary];
191  for (const auto &index_key: adj_[index_binary]) {
192  if (zero_count_[index_key] + binary == 1) {
193  val += poly_value_list_[index_key];
194  }
195  flag = true;
196  abs_val += std::abs(poly_value_list_[index_key]);
197  }
198  dE_[index_binary] = (-2*binary + 1)*val;
199 
200  if (flag && max_effective_dE_ < abs_val) {
201  max_effective_dE_ = abs_val;
202  }
203  }
204  }
205  else {
206  throw std::runtime_error("Unknown vartype detected");
207  }
208  }

212  void update_spin_system(const graph::Index index_update_spin) {
213  for (const auto &index_key: adj_[index_update_spin]) {
214  const FloatType val = 4.0*poly_value_list_[index_key];
215  const int8_t sign = sign_key_[index_key];
216  for (const auto &index_spin: poly_key_list_[index_key]) {
217  if (index_spin != index_update_spin) {
218  dE_[index_spin] += val*sign;
219  }
220  }
221  sign_key_[index_key] *= -1;
222  }
223  dE_[index_update_spin] *= -1;
224  variables[index_update_spin] *= -1;
225  }
226 
229  void update_binary_system(const graph::Index index_update_binary) {
230  const graph::Binary update_binary = variables[index_update_binary];
231  const int coeef = -2*update_binary + 1;
232  const int count = +2*update_binary - 1;
233  for (const auto &index_key: adj_[index_update_binary]) {
234  const FloatType val = poly_value_list_[index_key];
235  for (const auto &index_binary: poly_key_list_[index_key]) {
236  const graph::Binary binary = variables[index_binary];
237  if (zero_count_[index_key] + update_binary + binary == 2 && index_binary != index_update_binary) {
238  dE_[index_binary] += coeef*(-2*binary + 1)*val;
239  }
240  }
241  zero_count_[index_key] += count;
242  }
243  dE_[index_update_binary] *= -1;
244  variables[index_update_binary] = 1 - variables[index_update_binary];
245  }

349  void SetInteractions(const graph::Polynomial<FloatType> &poly_graph) {
350  const auto &poly_key_list = poly_graph.get_keys();
351  const auto &poly_value_list = poly_graph.get_values();
352 
353  if (poly_key_list.size() != poly_value_list.size()) {
354  throw std::runtime_error("The sizes of key_list and value_list must match each other");
355  }
356 
357  if (poly_key_list.size() == 0) {
358  throw std::runtime_error("The interaction is empty.");
359  }
360 
361  std::unordered_set<graph::Index> active_variable_set;
362 
363  poly_key_list_.clear();
364  poly_value_list_.clear();
365 
366  for (std::size_t i = 0; i < poly_key_list.size(); ++i) {
367  if (poly_value_list[i] != 0.0) {
368  poly_key_list_.push_back(poly_key_list[i]);
369  poly_value_list_.push_back(poly_value_list[i]);
370  for (const auto &it: poly_key_list[i]) {
371  active_variable_set.emplace(it);
372  }
373  }
374  }
375  num_interactions_ = static_cast<int64_t>(poly_key_list_.size());
376  active_variables_ = std::vector<graph::Index>(active_variable_set.begin(), active_variable_set.end());
377  std::sort(active_variables_.begin(), active_variables_.end());
378  }

381  void ResetZeroCount() {
382  if (vartype != cimod::Vartype::BINARY) {
383  return;
384  }
385  zero_count_.resize(num_interactions_);
386 #pragma omp parallel for
387  for (int64_t i = 0; i < num_interactions_; ++i) {
388  int64_t zero_count = 0;
389  for (const auto &index: poly_key_list_[i]) {
390  if (variables[index] == 0) {
391  zero_count++;
392  }
393  }
394  zero_count_[i] = zero_count;
395  }
396  }

399  void ResetSignKey() {
400  if (vartype != cimod::Vartype::SPIN) {
401  return;
402  }
403  sign_key_.resize(num_interactions_);
404 #pragma omp parallel for
405  for (int64_t i = 0; i < num_interactions_; ++i) {
406  int8_t sign_key = 1;
407  for (const auto &index: poly_key_list_[i]) {

430  std::pair<std::vector<graph::Index>, FloatType> FindMaxInteraction() const {
431  if (poly_key_list_.size() == 0) {
432  throw std::runtime_error("Interactions are empty.");
433  }
434  FloatType max_val = 0.0;
435  std::vector<graph::Index> max_key = {};
436  for (std::size_t i = 0; i < poly_key_list_.size(); ++i) {
437  if (std::abs(max_val) < std::abs(poly_value_list_[i])) {
438  max_val = poly_value_list_[i];
439  max_key = poly_key_list_[i];
440  }
441  }
442  return std::pair<std::vector<graph::Index>, FloatType>(max_key, max_val);
443  }

447  std::pair<std::vector<graph::Index>, FloatType> FindMinInteraction(const FloatType threshold = 0.0) const {
448  if (poly_key_list_.size() == 0) {
449  throw std::runtime_error("Interactions are empty.");
450  }
451  FloatType min_val = 0.0;
452  std::vector<graph::Index> min_key;
453 
454  // Set initial value larger than threshold.
455  bool flag_success_initialize = false;
456  for (std::size_t i = 0; i < poly_key_list_.size(); ++i) {
457  if (std::abs(threshold) < std::abs(poly_value_list_[i])) {
458  min_val = poly_value_list_[i];
459  min_key = poly_key_list_[i];
460  flag_success_initialize = true;
461  break;
462  }
463  }
464  if (!flag_success_initialize) {
465  std::stringstream ss;
466  ss << "No interactions larger than threshold=" << std::abs(threshold) << std::endl;
467  throw std::runtime_error(ss.str());
468  }
469 
470  for (std::size_t i = 0; i < poly_key_list_.size(); ++i) {
471  if (std::abs(threshold) < std::abs(poly_value_list_[i]) && std::abs(poly_value_list_[i]) < std::abs(min_val)) {
472  min_val = poly_value_list_[i];
473  min_key = poly_key_list_[i];
474  }
475  }
476 
477  if (std::abs(min_val) <= std::abs(threshold)) {
478  std::stringstream ss;
479  ss << "Unknown error in " << __func__ << std::endl;
480  throw std::runtime_error(ss.str());
481  }
482  return std::pair<std::vector<graph::Index>, FloatType>(min_key, min_val);
483  }
