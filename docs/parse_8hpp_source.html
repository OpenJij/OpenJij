44 inline auto json_parse(const json& obj, bool relabel=true){

50  if(relabel){
51  //re-labeling

63 template<typename FloatType>
64 inline auto json_parse_polynomial(const nlohmann::json& obj, const bool relabel = true) {
65 
66  if(obj.at("type") != "BinaryPolynomialModel") {
67  throw std::runtime_error("Type must be \"BinaryPolynomialModel\".\n");
68  }
69 
70  if (obj.at("poly_key_distance_list").size() != obj.at("poly_value_list").size()) {
71  throw std::runtime_error("The sizes of key_list and value_list must match each other");
72  }
73 
74  const std::size_t num_variables = obj["variables"].size();
75  const int64_t num_interactions = static_cast<int64_t>(obj["poly_value_list"].size());
76 
77  if (num_interactions == 0) {
78  throw std::runtime_error("The interaction is empty.");
79  }
80 
81  const cimod::PolynomialKeyList<std::size_t> &poly_key_distance_list = obj["poly_key_distance_list"];
82  const cimod::PolynomialValueList<FloatType> &poly_value_list = obj["poly_value_list"];
83  cimod::PolynomialKeyList<Index> poly_key_list(num_interactions);
84 
85  if (relabel) {
86  std::vector<std::size_t> sorted_variables;
87  sorted_variables.resize(num_variables);
88  std::iota(sorted_variables.begin(), sorted_variables.end(), 0);
89 #pragma omp parallel for
90  for (int64_t i = 0; i < num_interactions; ++i) {
91  std::vector<Index> temp;
92  for (const auto &it: poly_key_distance_list[i]) {
93  temp.push_back(sorted_variables[it]);
94  }
95  std::sort(temp.begin(), temp.end());
96  poly_key_list[i] = temp;
97  }
98  }
99  else {
100  const std::vector<Index> &variables = obj["variables"];
101 #pragma omp parallel for
102  for (int64_t i = 0; i < num_interactions; ++i) {
103  std::vector<Index> temp;
104  for (const auto &it: poly_key_distance_list[i]) {
105  temp.push_back(variables[it]);
106  }
107  std::sort(temp.begin(), temp.end());
108  poly_key_list[i] = temp;
109  }
110  }
111  return std::tuple<cimod::PolynomialKeyList<Index>, cimod::PolynomialValueList<FloatType>>(poly_key_list, poly_value_list);
112 }
