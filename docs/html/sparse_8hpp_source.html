245  FloatType energy(const Spins& spins) const{
246  if(!(spins.size() == this->get_num_spins())){
247  std::out_of_range("Out of range in energy in Sparse graph.");
248  }
249 
250  FloatType ret = 0;
251  for(std::size_t ind=0; ind<this->get_num_spins(); ind++){
252  for(auto& adj_ind : _list_adj_nodes[ind]){
253  if(ind != adj_ind)
254  ret += (1./2) * this->J(ind, adj_ind) * spins[ind] * spins[adj_ind];
255  else
256  ret += this->h(ind) * spins[ind];
257  }
258  }
259 
260  return ret;
261  }
