230  inline void initialize_gpu(const graph::Chimera<FloatType>& init_interaction, const TrotterSpins& trotter_spins, int gpu_num){
231 
232  //specify gpu number
233  HANDLE_ERROR_CUDA(cudaSetDevice(gpu_num));
234 
235  //generate temporary interaction and spin
236  const std::size_t localsize = info.rows*info.cols*info.chimera_unitsize;
237 
238  auto J_out_p = utility::cuda::make_host_unique<FloatType[]>(localsize);
239  auto J_out_n = utility::cuda::make_host_unique<FloatType[]>(localsize);
240  auto J_in_04 = utility::cuda::make_host_unique<FloatType[]>(localsize);
241  auto J_in_15 = utility::cuda::make_host_unique<FloatType[]>(localsize);
242  auto J_in_26 = utility::cuda::make_host_unique<FloatType[]>(localsize);
243  auto J_in_37 = utility::cuda::make_host_unique<FloatType[]>(localsize);
244  auto h = utility::cuda::make_host_unique<FloatType[]>(localsize);
245  auto temp_spin = utility::cuda::make_host_unique<int32_t[]>(localsize*info.trotters);
246 
247  using namespace chimera_cuda;
248 
249  //copy interaction info to std::vector variables
250  for(size_t r=0; r<info.rows; r++){
251  for(size_t c=0; c<info.cols; c++){
252  for(size_t i=0; i<info.chimera_unitsize; i++){
253 
254  J_out_p[glIdx(info,r,c,i)] = 0;
255  J_out_n[glIdx(info,r,c,i)] = 0;
256  J_in_04[glIdx(info,r,c,i)] = 0;
257  J_in_15[glIdx(info,r,c,i)] = 0;
258  J_in_26[glIdx(info,r,c,i)] = 0;
259  J_in_37[glIdx(info,r,c,i)] = 0;
260  h[glIdx(info,r,c,i)] = 0;
261 
262  if(r > 0 && i < 4){
263  //MINUS_R
264  J_out_p[glIdx(info,r,c,i)] = init_interaction.J(r,c,i,graph::ChimeraDir::MINUS_R);
265  }
266  if(c > 0 && 4 <= i){
267  //MINUS_C
268  J_out_p[glIdx(info,r,c,i)] = init_interaction.J(r,c,i,graph::ChimeraDir::MINUS_C);
269  }
270  if(r < info.rows-1 && i < 4){
271  //PLUS_R
272  J_out_n[glIdx(info,r,c,i)] = init_interaction.J(r,c,i,graph::ChimeraDir::PLUS_R);
273  }
274  if(c < info.cols-1 && 4 <= i){
275  //PLUS_C
276  J_out_n[glIdx(info,r,c,i)] = init_interaction.J(r,c,i,graph::ChimeraDir::PLUS_C);
277  }
278 
279  //inside chimera unit
280  J_in_04[glIdx(info,r,c,i)] = init_interaction.J(r,c,i,graph::ChimeraDir::IN_0or4);
281  J_in_15[glIdx(info,r,c,i)] = init_interaction.J(r,c,i,graph::ChimeraDir::IN_1or5);
282  J_in_26[glIdx(info,r,c,i)] = init_interaction.J(r,c,i,graph::ChimeraDir::IN_2or6);
283  J_in_37[glIdx(info,r,c,i)] = init_interaction.J(r,c,i,graph::ChimeraDir::IN_3or7);
284 
285  //local field
286  h[glIdx(info,r,c,i)] = init_interaction.h(r,c,i);
287  }
288  }
289  }
290 
291  //copy spin info to std::vector variables
292  for(size_t t=0; t<info.trotters; t++){
293  for(size_t r=0; r<info.rows; r++){
294  for(size_t c=0; c<info.cols; c++){
295  for(size_t i=0; i<info.chimera_unitsize; i++){
296  temp_spin[glIdx(info,r,c,i,t)] = trotter_spins[t][init_interaction.to_ind(r,c,i)];
297  }
298  }
299  }
300  }
301 
302  //cudaMemcpy
303  HANDLE_ERROR_CUDA(cudaMemcpy(interaction.J_out_p.get(), J_out_p.get(), localsize*sizeof(FloatType), cudaMemcpyHostToDevice));
304  HANDLE_ERROR_CUDA(cudaMemcpy(interaction.J_out_n.get(), J_out_n.get(), localsize*sizeof(FloatType), cudaMemcpyHostToDevice));
305  HANDLE_ERROR_CUDA(cudaMemcpy(interaction.J_in_04.get(), J_in_04.get(), localsize*sizeof(FloatType), cudaMemcpyHostToDevice));
306  HANDLE_ERROR_CUDA(cudaMemcpy(interaction.J_in_15.get(), J_in_15.get(), localsize*sizeof(FloatType), cudaMemcpyHostToDevice));
307  HANDLE_ERROR_CUDA(cudaMemcpy(interaction.J_in_26.get(), J_in_26.get(), localsize*sizeof(FloatType), cudaMemcpyHostToDevice));
308  HANDLE_ERROR_CUDA(cudaMemcpy(interaction.J_in_37.get(), J_in_37.get(), localsize*sizeof(FloatType), cudaMemcpyHostToDevice));
309  HANDLE_ERROR_CUDA(cudaMemcpy(interaction.h.get(), h.get(), localsize*sizeof(FloatType), cudaMemcpyHostToDevice));
310 
311  HANDLE_ERROR_CUDA(cudaMemcpy(spin.get(), temp_spin.get(), localsize*info.trotters*sizeof(int32_t), cudaMemcpyHostToDevice));
312  }
