115  inline void _checkpair(Index idx1, Index idx2) const{
116  std::int64_t r1,c1,r2,c2;
117  std::tie(r1,c1) = to_rc(idx1);
118  std::tie(r2,c2) = to_rc(idx2);
119  if(
120  !((r1 == r2 && std::abs(c1 - c2) == 1) || //horizontal connect
121  (c1 == c2 && std::abs(r1 - r2) == 1) || //vertical connect
122  (r1 == r2 && c1 == c2)) //local field
123  ){
124  throw std::runtime_error("invalid index pair "+std::to_string(idx1)+" "+std::to_string(idx2)+" inserted in Square");
125  }
126  }

139  Index to_ind(std::int64_t r, std::int64_t c) const{
140  if(!(-1 <= r && r <= (std::int64_t)_num_row)){
141  throw std::runtime_error("invalid value r="+std::to_string(r)+" inserted in Square");
142  }
143 
144  if(!(-1 <= c && c <= (std::int64_t)_num_column)){
145  throw std::runtime_error("invalid value c="+std::to_string(c)+" inserted in Square");
146  }
147 
148  return _num_column * mod_r(r) + mod_c(c);
149  }

175  Square(std::size_t num_row, std::size_t num_column, FloatType init_val=0)
176  : Sparse<FloatType>(num_row*num_column, 4+1), _init_val(init_val), _num_row(num_row), _num_column(num_column){
177  assert(num_row >= 1);
178  assert(num_column >= 1);
179 
180  for(std::size_t r=0; r<_num_row; r++){
181  for(std::size_t c=0; c<_num_column; c++){
182  //open boundary
183  if(r > 0){
184  //MINUS_R
185  this->Sparse<FloatType>::J(to_ind(r,c), to_ind(r-1,c)) = _init_val;
186  }
187  if(c > 0){
188  //MINUS_C
189  this->Sparse<FloatType>::J(to_ind(r,c), to_ind(r,c-1)) = _init_val;
190  }
191  if(r < num_row-1){
192  //PLUS_R
193  this->Sparse<FloatType>::J(to_ind(r,c), to_ind(r+1,c)) = _init_val;
194  }
195  if(c < num_column-1){
196  //PLUS_C
197  this->Sparse<FloatType>::J(to_ind(r,c), to_ind(r,c+1)) = _init_val;
198  }
199  //local field
200  this->Sparse<FloatType>::J(to_ind(r,c), to_ind(r,c)) = _init_val;
201  }
202  }
