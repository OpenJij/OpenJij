69  ContinuousTimeIsing(const SpinConfiguration& init_spin_config,
70  const GraphType& init_interaction,
71  const double gamma)
72  : spin_config(init_spin_config),
73  num_spins(init_spin_config.size()+1),
74  interaction(utility::gen_matrix_from_graph<Eigen::RowMajor>(init_interaction)),
75  gamma(gamma) {
76 
77  assert(init_spin_config.size() == init_interaction.get_num_spins());
78 
79  // insert numbers to diagnonal elements in the interaction
80  SparseMatrixXx diag(init_interaction.get_num_spins()+1, init_interaction.get_num_spins()+1);
81 
82  for(typename SparseMatrixXx::InnerIterator it(interaction, init_interaction.get_num_spins()); it; ++it) {
83  std::size_t j = it.index();
84  FloatType v = it.value();
85  if(j != init_interaction.get_num_spins()){
86  diag.insert(j,j) = v;
87  }
88  else{
89  diag.insert(j,j) = -1;
90  }
91  }
92 
93 
94  interaction += diag;
95 
96  spin_config.push_back(std::vector<CutPoint> { CutPoint(0.0, 1) });
97  // initialize auxiliary spin with 1 along entire timeline
98  }

175  size_t get_temporal_spin_index(graph::Index site_index, TimeType time_point) const {
176  static const auto first_lt = [](CutPoint x, CutPoint y) { return x.first < y.first; };
177  // function to compare two time points (lt; less than)
178 
179  const auto& timeline = this->spin_config[site_index];
180  const auto dummy_cut = CutPoint(time_point, 0); // dummy variable for binary search
181  auto found_itr = std::upper_bound(timeline.begin(),
182  timeline.end(),
183  dummy_cut,
184  first_lt);
185 
186  if(found_itr == timeline.begin()) { // if the time_point lies before any time points
187  found_itr = timeline.end() - 1; // periodic boundary condition
188  } else {
189  found_itr--;
190  }
191 
192  return std::distance(timeline.begin(), found_itr);
193  }
