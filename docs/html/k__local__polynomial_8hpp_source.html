129  void reset_binaries(const graph::Binaries &init_binaries) {
130  if (init_binaries.size() != binaries.size()) {
131  throw std::runtime_error("The size of initial binaries does not equal to system size");
132  }
133  for (const auto &index_binary: active_binaries_) {
134  if (binaries[index_binary] != init_binaries[index_binary]) {
135  update_system_single(index_binary);
136  }
137  if (binaries[index_binary] != init_binaries[index_binary]) {
138  std::stringstream ss;
139  ss << "Unknown error detected in " << __func__;
140  throw std::runtime_error(ss.str());
141  }
142  }

146  void reset_dE() {
147  dE_.clear();
148  dE_v_.clear();
149  dE_.resize(num_binaries);
150  dE_v_.resize(num_binaries);
151 
152  max_effective_dE = std::abs(poly_value_list_.front());
153  min_effective_dE = std::abs(poly_value_list_.front());
154 
155  for (const auto &index_binary: active_binaries_) {
156  FloatType val = 0.0;
157  FloatType abs_val = 0.0;
158  bool flag = false;
159  const graph::Binary binary = binaries[index_binary];
160  for (const auto &index_key: adj_[index_binary]) {
161  if (zero_count_[index_key] + binary == 1) {
162  val += poly_value_list_[index_key];
163  abs_val += std::abs(poly_value_list_[index_key]);
164  flag = true;
165  }
166  }
167  dE_[index_binary] = (-2*binary + 1)*val;
168  dE_v_[index_binary] = dE_[index_binary];
169 
170  if (flag && max_effective_dE < abs_val) {
171  max_effective_dE = abs_val;
172  }
173  if (flag && min_effective_dE > abs_val) {
174  min_effective_dE = abs_val;
175  }
176  }

219  void update_system_single(const graph::Index index_update_binary) {
220  const graph::Binary update_binary = binaries[index_update_binary];
221  const int coeef = -2*update_binary + 1;
222  const int count = +2*update_binary - 1;
223  for (const auto &index_key: adj_[index_update_binary]) {
224  const FloatType val = poly_value_list_[index_key];
225  for (const auto &index_binary: poly_key_list_[index_key]) {
226  const graph::Binary binary = binaries[index_binary];
227  if (zero_count_[index_key] + update_binary + binary == 2 && index_binary != index_update_binary) {
228  dE_[index_binary] += coeef*(-2*binary + 1)*val;
229  dE_v_[index_binary] = dE_[index_binary];
230  }
231  }
232  zero_count_[index_key] += count;
233  zero_count_v_[index_key] = zero_count_[index_key];
234  }
235  dE_[index_update_binary] *= -1;
236  dE_v_[index_update_binary] = dE_[index_update_binary];
237  binaries[index_update_binary] = 1 - binaries[index_update_binary];
238  binaries_v_[index_update_binary] = binaries[index_update_binary];
239  }

243  void virtual_update_system_single(const graph::Index index_update_binary) {
244  const graph::Binary update_binary = binaries_v_[index_update_binary];
245  const int coeef = -2*update_binary + 1;
246  const int count = +2*update_binary - 1;
247  for (const auto &index_key: adj_[index_update_binary]) {
248  const FloatType val = poly_value_list_[index_key];
249  for (const auto &index_binary: poly_key_list_[index_key]) {
250  const graph::Binary binary = binaries_v_[index_binary];
251  if (zero_count_v_[index_key] + update_binary + binary == 2 && index_binary != index_update_binary) {
252  dE_v_[index_binary] += coeef*(-2*binary + 1)*val;
253  update_index_dE_v_.emplace(index_binary);
254  }
255  }
256  zero_count_v_[index_key] += count;
257  update_index_zero_count_v_.push_back(index_key);
258  }
259  dE_v_[index_update_binary] *= -1;
260  update_index_dE_v_.emplace(index_update_binary);
261  binaries_v_[index_update_binary] = 1 - binaries_v_[index_update_binary];
262  update_index_binaries_v_.push_back(index_update_binary);
263  }

444  void SetAdj() {
445  adj_.clear();
446  adj_.resize(num_binaries);
447  for (int64_t i = 0; i < num_interactions_; ++i) {
448  for (const auto &index: poly_key_list_[i]) {

453  //sort

455 
456  int64_t adj_size = static_cast<int64_t>(adj_.size());
457 #pragma omp parallel for
458  for (int64_t i = 0; i < adj_size; ++i) {
459  std::sort(adj_[i].begin(), adj_[i].end(), compare);

461  }

464  void ResetZeroCount() {
465  zero_count_.resize(num_interactions_);
466  zero_count_v_.resize(num_interactions_);
467 #pragma omp parallel for
468  for (int64_t i = 0; i < num_interactions_; ++i) {
469  int64_t zero_count = 0;
470  for (const auto &index: poly_key_list_[i]) {
471  if (binaries[index] == 0) {
472  zero_count++;
473  }
474  }
475  zero_count_[i] = zero_count;
476  zero_count_v_[i] = zero_count;
477  }
